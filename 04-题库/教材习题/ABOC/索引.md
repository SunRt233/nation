```dataview
TABLE WITHOUT ID
	link(file.path, title) AS 题目,
	difficulty AS 难度,
	question_type AS 题型

WHERE type = "题目"
  AND difficulty >= 4
  AND contains(knowledge_points, "Diels-Alder反应")
SORT difficulty DESC
```
