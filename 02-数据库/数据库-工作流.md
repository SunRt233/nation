## 一、🔥 高优 Follow-up（辅助镜像）

> 自动聚合 `00-首页/活跃任务/` 中未完成的高优任务卡。

```dataview
TABLE WITHOUT ID
  file.link as 任务,
  area as 领域,
  join(evidence, " · ") as 证据,
  updated as 更新
FROM "00-首页/活跃任务"
WHERE type = "活跃任务卡"
  AND status = "active"
  AND priority = "P1"
SORT updated DESC
LIMIT 10
```

## 二、📋 待填充知识点（辅助监控）

> 状态为「骨架」或「初稿」的知识点，是需要继续填充的内容缺口。

```dataview
TABLE WITHOUT ID
  link(file.name, title) as 知识点,
  subject as 模块,
  status as 状态,
  importance as 重要度,
  difficulty as 难度
FROM "03-知识点"
WHERE type = "知识点"
  AND status != "已填充"
SORT subject ASC, importance DESC, difficulty DESC
```

## 三、📝 真题数据质量待修复

### 3.1 缺少 knowledge_points（影响 Dataview 关联）

```dataview
TABLE WITHOUT ID
  link(file.name, title) as 题目,
  year as 年份,
  module as 模块
FROM "04-题库/真题"
WHERE !knowledge_points OR length(knowledge_points) = 0
SORT year DESC, file.name ASC
```

### 3.2 缺少 syllabus_codes（影响考纲关联）

```dataview
TABLE WITHOUT ID
  link(file.name, title) as 题目,
  year as 年份,
  module as 模块
FROM "04-题库/真题"
WHERE !syllabus_codes OR length(syllabus_codes) = 0
```

## 四、📚 资料提炼进度跟踪

> 监控「计划提取题数」vs「实际提取题数」的进度。

```dataview
TABLE WITHOUT ID
  link(file.name, title) as 提炼页,
  source_type as 来源类型,
  year as 年份,
  review_status as 审核状态,
  problems_planned as 计划题数,
  problems_extracted as 已提取,
  round(problems_extracted / problems_planned * 100, 1) + "%" as 进度,
  join(knowledge_points, "、") as 涉及知识点
FROM "07-资料提炼"
SORT review_status ASC, file.mtime DESC
```

## 六、⚠️ 踩坑速查（最近 3 条）

> 自动抽取 [[经验沉淀]] 中标了 🔥 的 H3 条目。

```dataviewjs
const exp = await dv.io.load("00-首页/经验沉淀.md");
const lines = exp.split("\n");

const tops = [];
let currentCategory = "";
for (const line of lines) {
  const cat = line.match(/^##\s+([一二三四五六七八九十]+)、(.+?)\s*$/);
  if (cat) {
    currentCategory = cat[2].trim();
    continue;
  }
  const m = line.match(/^###\s+(\S+)\s+(.+?)\s*🔥\s*$/);
  if (m) {
    tops.push({ code: m[1], title: m[2].trim(), category: currentCategory });
  }
}

if (tops.length === 0) {
  dv.paragraph("⚠️ 未在 [[经验沉淀]] 中找到 🔥 标记的条目。");
} else {
  dv.table(
    ["编号", "问题", "分类"],
    tops.slice(0, 3).map(t => [t.code, t.title, t.category])
  );
}
```
