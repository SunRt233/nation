## 交互组卷面板

> 直接在本页输入关键词即可筛题。支持：
> - 多关键词：用空格分隔
> - 模块筛选：全部 / 四大模块
> - 难度筛选：基础 / 综合 / 挑战
> - 轮次预设
> - 查询模板复制
> - 当前候选题清单导出

```dataviewjs
const root = dv.el("div", "", { cls: "dv-pack-panel" });

function makeCheck(label, checked = true) {
  const wrap = document.createElement("label");
  wrap.style.display = "flex";
  wrap.style.alignItems = "center";
  wrap.style.gap = "6px";
  const input = document.createElement("input");
  input.type = "checkbox";
  input.checked = checked;
  wrap.appendChild(input);
  wrap.appendChild(document.createTextNode(label));
  bucketWrap.appendChild(wrap);
  return input;
}

const baseCheck = makeCheck("基础（difficulty 2）", true);
const midCheck = makeCheck("综合（difficulty 3）", true);
const highCheck = makeCheck("挑战（difficulty 4-5）", true);

const presetRow = document.createElement("div");
presetRow.style.display = "flex";
presetRow.style.alignItems = "end";
presetRow.style.gap = "12px";

const presetWrap = document.createElement("label");
presetWrap.style.display = "flex";

presetWrap.style.gap = "4px";
presetWrap.textContent = "轮次预设";
const presetSelect = document.createElement("select");
[
  "自定义",
  "第一轮（基础班）",
  "第二轮（提高班）",
  "第三轮（深化）",
  "第四轮（冲刺）"
].forEach(name => {
  const opt = document.createElement("option");

  opt.textContent = name;

copyBtn.textContent = "复制当前查询模板";

copyBtn.style.borderRadius = "6px";

presetRow.appendChild(copyBtn);

const actionMsg = document.createElement("span");
actionMsg.style.color = "var(--text-muted)";
actionMsg.style.fontSize = "0.9em";
presetRow.appendChild(actionMsg);

const hint = document.createElement("p");
hint.style.marginTop = "0";
hint.style.color = "var(--text-muted)";
hint.textContent = "提示：关键词会在 knowledge_points、标题、来源中做模糊匹配。";
root.appendChild(hint);

const summary = document.createElement("div");
summary.style.margin = "8px 0 12px";
root.appendChild(summary);

root.appendChild(resultMount);

const pages = dv.pages('"04-题库/真题"').where(p => p.type == "题目").array();

function difficultyMatch(p) {
  const d = Number(p.difficulty ?? 0);
  return (baseCheck.checked && d === 2)
    || (midCheck.checked && d === 3)
    || (highCheck.checked && d >= 4);
}

function moduleMatch(p) {
  return moduleSelect.value === "全部模块" || p.module === moduleSelect.value;
}

function keywordMatch(p, keywords) {
  if (!keywords.length) return true;
  const haystack = [
    p.title ?? "",
    p.module ?? "",
    p.question_type ?? "",
    p.source ?? "",
    Array.isArray(p.knowledge_points) ? p.knowledge_points.join(" ") : String(p.knowledge_points ?? "")
  ].join(" ").toLowerCase();
  return keywords.every(k => haystack.includes(k));
}

function bucketLabel(d) {
  if (d === 2) return "基础";
  if (d === 3) return "综合";
  if (d >= 4) return "挑战";
  return "其他";
}

function presetConfig(name) {
  if (name === "第一轮（基础班）") {
    return { limit: 10, base: true, mid: true, high: false, hint: "建议选 3-4 道基础 + 1-2 道综合" };
  }
  if (name === "第二轮（提高班）") {
    return { limit: 14, base: true, mid: true, high: true, hint: "建议选 2 道基础 + 2-3 道综合 + 1 道挑战" };
  }
  if (name === "第三轮（深化）") {
    return { limit: 16, base: false, mid: true, high: true, hint: "建议选 1 道基础（可选） + 2 道综合 + 2 道挑战" };
  }
  if (name === "第四轮（冲刺）") {
    return { limit: 16, base: false, mid: true, high: true, hint: "建议偏重挑战题，综合题用于热身与查漏" };
  }
  return { limit: Number(limitInput.value || 24), base: baseCheck.checked, mid: midCheck.checked, high: highCheck.checked, hint: "当前为自定义模式" };
}

function setMessage(text) {
  actionMsg.textContent = text;
  if (!text) return;
  clearTimeout(setMessage._timer);
  setMessage._timer = setTimeout(() => {
    actionMsg.textContent = "";
  }, 2500);
}

function buildTemplate(keywords) {
  const words = keywords.length ? keywords : ["目标KP"];
  const wherePart = words
    .map(k => `contains(string(knowledge_points), "${k}")`)
    .join("\n    OR ");
  return [
    "```dataview",
    "TABLE WITHOUT ID",
    '  link(file.path, title) AS 题目,',
    '  year AS 年份,',
    '  module AS 模块,',
    '  question_type AS 题型,',
    '  difficulty AS 难度,',
    '  join(knowledge_points, "、") AS 知识点',
    'FROM "04-题库/真题"',
    "WHERE (",
    `    ${wherePart}`,
    "  )",
    "  AND difficulty >= 2",
    "  AND difficulty <= 4",
    "SORT difficulty ASC, year DESC",
    "LIMIT 20",
    "```"
  ].join("\n");
}

function render() {
  resultMount.innerHTML = "";
  exportMount.innerHTML = "";
  const keywords = keywordInput.value.trim().toLowerCase().split(/\s+/).filter(Boolean);
  const limit = Math.max(5, Math.min(60, Number(limitInput.value || 24)));

  const rows = pages
    .filter(p => difficultyMatch(p))
    .filter(p => moduleMatch(p))
    .filter(p => keywordMatch(p, keywords))
    .sort((a, b) => {
      const da = Number(a.difficulty ?? 0);
      const db = Number(b.difficulty ?? 0);
      if (da !== db) return da - db;
      return Number(b.year ?? 0) - Number(a.year ?? 0);
    })
    .slice(0, limit);

  const counts = { 基础: 0, 综合: 0, 挑战: 0 };
  rows.forEach(p => {
    const label = bucketLabel(Number(p.difficulty ?? 0));
    if (counts[label] != null) counts[label]++;
  });

  const preset = presetConfig(presetSelect.value);
  summary.innerHTML = `命中 <strong>${rows.length}</strong> 题`
    + ` · 基础 ${counts["基础"]}`
    + ` · 综合 ${counts["综合"]}`
    + ` · 挑战 ${counts["挑战"]}`
    + ` · 预设建议：${preset.hint}`;

  if (!rows.length) {
    const empty = document.createElement("p");
    empty.textContent = "没有匹配结果。建议减少关键词，或放宽模块/难度限制。";
    resultMount.appendChild(empty);
    return;
  }

  const overallTitle = document.createElement("p");
  overallTitle.innerHTML = "<strong>总览表</strong>";
  resultMount.appendChild(overallTitle);

  dv.table(
    ["层级", "题目", "年份", "模块", "题型", "难度", "知识点"],
    rows.map(p => [
      bucketLabel(Number(p.difficulty ?? 0)),
      dv.fileLink(p.file.path, false, p.title),
      p.year ?? "",
      p.module ?? "",
      p.question_type ?? "",
      p.difficulty ?? "",
      Array.isArray(p.knowledge_points) ? p.knowledge_points.join("、") : (p.knowledge_points ?? "")
    ]),
    resultMount
  );

  const groups = [
    { label: "基础", match: (d) => d === 2 },
    { label: "综合", match: (d) => d === 3 },
    { label: "挑战", match: (d) => d >= 4 }
  ];

  for (const group of groups) {
    const subset = rows.filter(p => group.match(Number(p.difficulty ?? 0)));
    if (!subset.length) continue;

    const titleEl = document.createElement("p");
    titleEl.style.marginTop = "18px";
    titleEl.innerHTML = `<strong>${group.label}候选题</strong>（${subset.length} 题）`;
    resultMount.appendChild(titleEl);

    dv.table(
      ["题目", "年份", "模块", "题型", "难度", "知识点"],
      subset.map(p => [
        dv.fileLink(p.file.path, false, p.title),
        p.year ?? "",
        p.module ?? "",
        p.question_type ?? "",
        p.difficulty ?? "",
        Array.isArray(p.knowledge_points) ? p.knowledge_points.join("、") : (p.knowledge_points ?? "")
      ]),
      resultMount
    );
  }

  const exportTitle = document.createElement("p");
  exportTitle.style.marginTop = "18px";
  exportTitle.innerHTML = "<strong>导出当前候选题清单</strong>";
  exportMount.appendChild(exportTitle);

  const exportLines = rows.map((p, idx) => {
    const kpText = Array.isArray(p.knowledge_points) ? p.knowledge_points.join("、") : (p.knowledge_points ?? "");
    return `${idx + 1}. [${bucketLabel(Number(p.difficulty ?? 0))}] ${p.title}｜${p.year ?? ""}｜${p.module ?? ""}｜${p.question_type ?? ""}｜难度 ${p.difficulty ?? ""}${kpText ? `｜${kpText}` : ""}`;
  });

  const exportBox = document.createElement("textarea");
  exportBox.value = exportLines.join("\n");
  exportBox.rows = Math.min(Math.max(rows.length + 2, 8), 20);
  exportBox.style.width = "100%";
  exportBox.style.padding = "8px 10px";
  exportBox.style.border = "1px solid var(--background-modifier-border)";
  exportBox.style.borderRadius = "6px";
  exportBox.style.fontFamily = "var(--font-monospace)";
  exportBox.style.fontSize = "0.9em";
  exportMount.appendChild(exportBox);

  const exportBtn = document.createElement("button");
  exportBtn.textContent = "复制当前候选题清单";
  exportBtn.style.marginTop = "8px";
  exportBtn.style.padding = "6px 10px";
  exportBtn.style.border = "1px solid var(--background-modifier-border)";
  exportBtn.style.borderRadius = "6px";
  exportBtn.style.cursor = "pointer";
  exportMount.appendChild(exportBtn);

  exportBtn.addEventListener("click", async () => {
    try {
      await navigator.clipboard.writeText(exportBox.value);
      setMessage("已复制候选题清单");
    } catch {
      exportBox.select();
      setMessage("剪贴板失败，已选中文本");
    }
  });
}

[keywordInput, moduleSelect, limitInput, baseCheck, midCheck, highCheck].forEach(el => {
  el.addEventListener("input", render);
  el.addEventListener("change", render);
});

applyPresetBtn.addEventListener("click", () => {
  const preset = presetConfig(presetSelect.value);
  limitInput.value = preset.limit;
  baseCheck.checked = preset.base;
  midCheck.checked = preset.mid;
  highCheck.checked = preset.high;
  render();
  setMessage(`已应用：${presetSelect.value}`);
});

copyBtn.addEventListener("click", async () => {
  const keywords = keywordInput.value.trim().split(/\s+/).filter(Boolean);
  const text = buildTemplate(keywords);
  try {
    await navigator.clipboard.writeText(text);
    setMessage("已复制查询模板");
  } catch {
    setMessage("复制失败，请手动选取");
  }
});

keywordInput.value = "目标KP";
render();
